def reward_function(params):
  # Get the waypoints for the current track
  waypoints = params['waypoints']
  
  # Get the index of the closest waypoint in front of the vehicle
  closest_waypoints = params['closest_waypoints']
  closest_waypoint_index = closest_waypoints[1]
  
  # Calculate the distance between the vehicle and the racing line
  distance_from_center = params['distance_from_center']
  track_width = params['track_width']
  distance_from_racing_line = distance_from_center / (track_width / 2.0)
  
  # Calculate the reward based on the distance from the racing line
  if distance_from_racing_line > 0.0:
    reward = 1.0 / (distance_from_racing_line + 1.0)
  else:
    reward = 1.0
  
  return reward
